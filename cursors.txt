//CURSORS//
DECLARE
ID EMPLOYEE.EID%TYPE
NAME EMPLOYEE.ENAME%TYPE
CURSOR MYCURSOR IS
SELECT EID,ENAME FROM EMPLOYEE;


BEGIN
OPEN MYCURSOR;
LOOP
FETCH MYCURSOR INTO ID,NAME;
EXIT WHEN MYCURSOR%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(ID || ':'||NAME);
END LOOP;
CLOSE MYCURSOR;
END;
/


//FUNCTIONS//
LOWER('NAME');
UPPER('NAME');
LENGTH('STRING');
REPLACE('RED','R','B');
